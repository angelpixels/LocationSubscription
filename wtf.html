        let total = 0;

        const totalDisplay = document.querySelector('.total');


        function updateTotal() {

            totalDisplay.textContent = `Total based on your selections: $${total}`;

        }


        document.querySelectorAll('.subscription').forEach(sub => {

            sub.addEventListener('click', function () {

                document.querySelectorAll('.subscription').forEach(s => s.classList.remove('selected'));

                sub.classList.add('selected');

                total = parseInt(sub.getAttribute('data-price'));

                document.querySelectorAll('.addon:checked').forEach(addon => {

                    total += parseInt(addon.getAttribute('data-price'));

                });

                updateTotal();

            });

        });


        document.querySelectorAll('.addon').forEach(addon => {

            addon.addEventListener('change', function () {

                const price = parseInt(addon.getAttribute('data-price'));

                if (addon.checked) {

                    total += price;

                } else {

                    total -= price;

                }

                updateTotal();

            });

        });


        updateTotal();

    </script>
